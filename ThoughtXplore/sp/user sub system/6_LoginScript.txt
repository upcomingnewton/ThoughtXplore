CREATE OR REPLACE FUNCTION user_login
(
	emailid VARCHAR,
	password VARCHAR,
	login_type VARCHAR,
	ip VARCHAR,
	OUT userid INTEGER,
	OUT user_status INTEGER,
	OUT loginid INTEGER,
	OUT groupid INTEGER,
	OUT result INTEGER
) 
RETURNS RECORD AS $$
DECLARE
	user__id INTEGER;
	varrec RECORD;
	varrec1 RECORD;
	login_type_int INTEGER;
	temp INTEGER;
	--tmp TIMESTAMP WITH TIME ZONE DEFAULT now();
BEGIN 
--RAISE TIME 'time now is %',tmp;
IF NOT EXISTS ( SELECT id FROM "txUser_user" WHERE "UserEmail"=emailid ) THEN
	result := -651;
	groupid := -1;
	loginid := -1;
	user_status := -1;
	userid := -1;
	RETURN;
END IF;

IF NOT EXISTS ( SELECT id FROM "txUser_user" WHERE "UserEmail"=emailid AND "UserPassword"=password ) THEN
	result := -652;
	groupid := -1;
	loginid := -1;
	user_status := -1;
	userid := -1;
	RETURN;
END IF;

SELECT id INTO user__id FROM "txUser_user" WHERE "UserEmail"=emailid;
SELECT * INTO varrec FROM  checkstatus('txUser_user','ACTIVE',user__id);
user_status := varrec.scid;
IF varrec.result < 0 THEN 
	result := varrec.result;
	userid := user__id;	
	loginid := -660;
	SELECT "Group_id" INTO groupid FROM "txUser_usergroup" WHERE "User_id"=user__id;
	RETURN;
END IF;

SELECT id INTO login_type_int FROM "txUser_logintype" WHERE "LoginTypeName"=login_type;
IF login_type_int IS NULL THEN 
	result := -653;
	userid := user__id;	
	loginid := -1;
	SELECT "Group_id" INTO groupid FROM "txUser_usergroup" WHERE "User_id"=user__id;
	RETURN;
END IF;


SELECT * INTO varrec1 FROM  checkstatus('txUser_logintype','ACTIVE',login_type_int);
IF varrec1.result < 0 THEN 
	result := varrec.result1;
	userid := user__id;	
	loginid := -661;
	SELECT "Group_id" INTO groupid FROM "txUser_usergroup" WHERE "User_id"=user__id;
	RETURN;
END IF;

INSERT INTO "txUser_userloginlog"(user_id,"Login_From_id","LoginTime","LoginIP","LogoutTime")
VALUES(user__id,login_type_int,now(),ip,now());
GET DIAGNOSTICS temp := ROW_COUNT;
IF temp <> 1 THEN 
	result := -654;
	user_status := varrec.scid;
	userid := user__id;	
	loginid := -1;
	SELECT "Group_id" INTO groupid FROM "txUser_usergroup" WHERE "User_id"=user__id;
	RETURN;
ELSE
	result := 650;
	user_status := varrec.scid;
	userid := user__id;	
	SELECT id INTO loginid FROM "txUser_userloginlog" WHERE user_id=user__id AND "LoginIP"=ip;
	SELECT "Group_id" INTO groupid FROM "txUser_usergroup" WHERE "User_id"=user__id;
	RETURN;
END IF;
RETURN;
END;
$$ LANGUAGE plpgsql;
