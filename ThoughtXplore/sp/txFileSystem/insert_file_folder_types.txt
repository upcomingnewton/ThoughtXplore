/*insert a new folder-type*/

CREATE OR REPLACE FUNCTION insert_foldertype(type_name text, type_desc text) RETURNS VOID AS $$
  
DECLARE
  id INTEGER;  
BEGIN 
 id=0;
 SELECT max(t."id") INTO id from "txFileSystem_foldertypes" t;
 IF id<>0 THEN
 	id=id+1;
 ELSE
    id= 1;
 END IF;
 INSERT INTO "txFileSystem_foldertypes" values (id,type_name, type_desc); 
 
END;
$$ LANGUAGE plpgsql;
